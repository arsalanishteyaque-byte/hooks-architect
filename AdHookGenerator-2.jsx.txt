export default function AdHookGenerator() {
  const [product, setProduct] = useState("");
  const [audience, setAudience] = useState("");
  const [platform, setPlatform] = useState("Meta Ads");
  const [tone, setTone] = useState("Curiosity");
  const [hooks, setHooks] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState("");
  const [usedCount, setUsedCount] = useState(0);
  const [showPaywall, setShowPaywall] = useState(false);
  const [selectedPlan, setSelectedPlan] = useState("Pro");
  const [toast, setToast] = useState("");

  const showToast = (msg) => {
    setToast(msg);
    setTimeout(() => setToast(""), 2000);
  };

  const copyHook = (text) => {
    navigator.clipboard.writeText(text).then(() => showToast("Copied!"));
  };

  const generate = async () => {
    if (usedCount >= FREE_LIMIT) {
      setShowPaywall(true);
      return;
    }
    if (!product.trim() || !audience.trim()) {
      setError("Please fill in your product and target audience.");
      return;
    }
    setError("");
    setLoading(true);
    setHooks([]);

    const prompt = `You are an expert Meta Ads copywriter specializing in scroll-stopping ad hooks.

Generate 5 powerful ad hooks for the following:
- Product/Service: ${product}
- Target Audience: ${audience}
- Platform: ${platform}
- Tone/Angle: ${tone}

Rules:
- Each hook must be 1-2 sentences max
- Make them highly specific, emotionally charged, and curiosity-driven
- Avoid generic phrases like "Are you tired of..." — be creative and bold
- Each hook should use a different psychological trigger (fear, desire, curiosity, social proof, urgency)

Return ONLY a JSON array of 5 strings. No explanation, no markdown, no preamble. Just the array.
Example: ["Hook 1", "Hook 2", "Hook 3", "Hook 4", "Hook 5"]`;

    try {
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          messages: [{ role: "user", content: prompt }],
        }),
      });

      const data = await response.json();
      const text = data.content?.map(i => i.text || "").join("") || "";
      const clean = text.replace(/```json|```/g, "").trim();
      const parsed = JSON.parse(clean);

      setHooks(parsed);
      const newCount = usedCount + 1;
      setUsedCount(newCount);
      if (newCount >= FREE_LIMIT) {
        setTimeout(() => setShowPaywall(true), 1200);
      }
    } catch (err) {
      setError("Something went wrong. Please try again.");
    } finally {
      setLoading(false);
    }
  };

  const creditsLeft = FREE_LIMIT - usedCount;

  return (
    <>
      <style>{styleTag}</style>
      <div className="app">
        <div className="bg-glow" />
        <div className="container">
          {/* Brand Header */}
          <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: "32px" }}>
            <div style={{ display: "flex", alignItems: "center", gap: "10px" }}>
              <div style={{
                width: "36px", height: "36px", borderRadius: "10px",
                background: "linear-gradient(135deg, #f97316, #f59e0b)",
                display: "flex", alignItems: "center", justifyContent: "center",
                fontSize: "18px", fontWeight: "800", color: "#000",
                fontFamily: "'Syne', sans-serif"
              }}>H</div>
              <div>
                <div style={{ fontFamily: "'Syne', sans-serif", fontWeight: "800", fontSize: "15px", color: "#fff", lineHeight: 1.1 }}>Hooks Architect</div>
                <div style={{ fontSize: "10px", color: "#f97316", fontWeight: "600", letterSpacing: "0.08em", textTransform: "uppercase" }}>Made for Marketing</div>
              </div>
            </div>
            <div className="badge" style={{ margin: 0 }}>
              <div className="dot" />
              AI-Powered
            </div>
          </div>

          <h1>Ad <span>Hook</span><br />Generator</h1>
          <p className="subtitle">Generate scroll-stopping first lines for your Meta, Instagram & Google ads in seconds.</p>

          {/* Credits bar */}
          <div className="credits-bar">
            <span className="credits-label">Free credits remaining</span>
            <div className="credits-dots">
              {Array.from({ length: FREE_LIMIT }).map((_, i) => (
                <div
                  key={i}
                  className={`credit-dot ${i < usedCount ? "used" : "left"}`}
                />
              ))}
            </div>
          </div>

          {/* Form */}
          <div className="form-card">
            <div className="field">
              <label>Product / Service</label>
              <input
                type="text"
                placeholder="e.g. Ayurvedic weight loss supplement"
                value={product}
                onChange={e => setProduct(e.target.value)}
              />
            </div>
            <div className="field">
              <label>Target Audience</label>
              <input
                type="text"
                placeholder="e.g. Women 25-45 struggling with belly fat"
                value={audience}
                onChange={e => setAudience(e.target.value)}
              />
            </div>
            <div className="field">
              <label>Platform</label>
              <select value={platform} onChange={e => setPlatform(e.target.value)}>
                <option>Meta Ads</option>
                <option>Instagram Reels</option>
                <option>Google Ads</option>
                <option>YouTube Ads</option>
              </select>
            </div>
            <div className="field">
              <label>Hook Angle</label>
              <select value={tone} onChange={e => setTone(e.target.value)}>
                <option>Curiosity</option>
                <option>Fear / Pain Point</option>
                <option>Desire / Dream Outcome</option>
                <option>Social Proof</option>
                <option>Shock / Controversy</option>
                <option>Urgency / Scarcity</option>
              </select>
            </div>

            <button
              className="generate-btn"
              onClick={generate}
              disabled={loading}
            >
              {loading ? "Generating..." : usedCount >= FREE_LIMIT ? "🔒 Unlock to Generate" : "⚡ Generate Hooks"}
            </button>

            {error && <div className="error-msg">{error}</div>}
          </div>

          {/* Results */}
          {loading && (
            <div className="loading">
              <div className="spinner" />
              Writing your hooks...
            </div>
          )}

          {hooks.length > 0 && !loading && (
            <div className="results">
              <div className="results-header">✦ Your hooks are ready</div>
              {hooks.map((hook, i) => (
                <div key={i} className="hook-card" onClick={() => copyHook(hook)}>
                  <div className="hook-number">{i + 1}</div>
                  <div className="hook-text">{hook}</div>
                  <button className="copy-btn" title="Copy">⧉</button>
                </div>
              ))}
            </div>
          )}
        </div>

        {/* Footer */}
        <div style={{
          textAlign: "center", padding: "20px", fontSize: "12px",
          color: "#444", borderTop: "1px solid rgba(255,255,255,0.05)",
          marginTop: "20px"
        }}>
          Made with ⚡ by <span style={{ color: "#f97316", fontWeight: "600" }}>Hooks Architect</span> · Made for Marketing
        </div>

        {/* Paywall */}
        {showPaywall && (
          <PaywallModal
            onClose={() => setShowPaywall(false)}
            selectedPlan={selectedPlan}
            setSelectedPlan={setSelectedPlan}
          />
        )}

        {/* Toast */}
        {toast && <div className="copy-toast">{toast}</div>}
      </div>
    </>
  );
}
